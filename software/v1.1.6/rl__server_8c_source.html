262 
263  // quit, if data not requested or not running or web disabled
264  if (status.state != RL_RUNNING || status.sampling == SAMPLING_OFF ||
265  status.conf.enable_web_server == 0 || get_data == 0) {
266  // print request id and status

309  }
310
